IOFileFilter dirFilter = FileFilterUtils.and(directories, visible);                                  |

observer.addListener(listener);                                                                      |                                                                                                     
try {                                                                                                |                                                                                                     
observer.initialize();                                                                               |

fail("Observer init() threw " + e);                                                                  |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |
