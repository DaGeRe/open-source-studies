                                                                                                     | /**                                                                                                 
                                                                                                     | * &lt;p&gt;Converts a String to upper case as per {@link String#toUpperCase(Locale)}.&lt;/p&gt;     
                                                                                                     | *                                                                                                   
                                                                                                     | * &lt;p&gt;A {@code null} input String returns {@code null}.&lt;/p&gt;                              
                                                                                                     | *                                                                                                   
                                                                                                     | * &lt;pre&gt;                                                                                       
                                                                                                     | * StringUtils.upperCase(null, Locale.ENGLISH)  = null                                               
                                                                                                     | * StringUtils.upperCase("", Locale.ENGLISH)    = ""                                                 
                                                                                                     | * StringUtils.upperCase("aBc", Locale.ENGLISH) = "ABC"                                              
                                                                                                     | * &lt;/pre&gt;                                                                                      
                                                                                                     | *                                                                                                   
                                                                                                     | * @param seq  the CharSequence to upper case, may be null                                           
                                                                                                     | * @param locale  the locale that defines the case transformation rules, must not be null            
                                                                                                     | * @return the upper cased String, {@code null} if null String input                                 
                                                                                                     | * @since 2.5                                                                                        
                                                                                                     | */                                                                                                  
                                                                                                     | public static String upperCase(CharSequence seq, Locale locale) {                                   
                                                                                                     | if (seq == null) {                                                                                  
                                                                                                     | return null;                                                                                        
                                                                                                     | }                                                                                                   
                                                                                                     | return seq.toString().toUpperCase(locale);                                                          
                                                                                                     | }
