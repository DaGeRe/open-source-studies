/**                                                                                                  |                                                                                                     
* &lt;p&gt;Returns a String representation in the given format.&lt;/p&gt;                            |                                                                                                     
*                                                                                                    |                                                                                                     
* &lt;p&gt;The format specified uses the syntax from {@link Formatter}.                              |                                                                                                     
* There are three arguments available:&lt;/p&gt;                                                     |                                                                                                     
* &lt;ol&gt;                                                                                         |                                                                                                     
* &lt;li&gt;The simple class name&lt;/li&gt;                                                         |                                                                                                     
* &lt;li&gt;The left object&lt;/li&gt;                                                               |                                                                                                     
* &lt;li&gt;The right object&lt;/li&gt;                                                              |                                                                                                     
* &lt;/ol&gt;                                                                                        |                                                                                                     
*                                                                                                    |                                                                                                     
* @param format  the format suitable for use with {@code Formatter}, not null                        |                                                                                                     
* @return a string describing for this object, not null                                              |                                                                                                     
*/                                                                                                   |                                                                                                     
public String toString(String format) {                                                              |                                                                                                     
return String.format(format, getClass().getSimpleName(), getLeft(), getRight());                     |                                                                                                     
}                                                                                                    |
