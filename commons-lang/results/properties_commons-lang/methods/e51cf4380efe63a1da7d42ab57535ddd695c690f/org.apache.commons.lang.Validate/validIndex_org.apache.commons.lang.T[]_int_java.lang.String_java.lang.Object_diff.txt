                                                                                                     | /**                                                                                                 
                                                                                                     | * &lt;p&gt;Validates that the index is within the bounds of the argument                            
                                                                                                     | * array; otherwise throwing an exception with the specified message.&lt;/p&gt;                      
                                                                                                     | *                                                                                                   
                                                                                                     | * &lt;pre&gt;Validate.validIndex(myArray, 2, "The array index is invalid: ");&lt;/pre&gt;           
                                                                                                     | *                                                                                                   
                                                                                                     | * &lt;p&gt;If the array is &lt;code&gt;null&lt;/code&gt;, then the message of the exception         
                                                                                                     | * is &quot;The validated object is null&quot;.&lt;/p&gt;                                            
                                                                                                     | *                                                                                                   
                                                                                                     | * @param &lt;T&gt; the array type                                                                   
                                                                                                     | * @param array the array to check                                                                   
                                                                                                     | * @param index the index                                                                            
                                                                                                     | * @param message the exception message if invalid                                                   
                                                                                                     | * @return the validated array (never &lt;code&gt;null&lt;/code&gt; for method chaining)             
                                                                                                     | * @throws NullPointerException if the array is &lt;code&gt;null&lt;/code&gt;                        
                                                                                                     | * @throws IndexOutOfBoundsException if the index is invalid                                         
                                                                                                     | * @see #validIndex(Object[], int)                                                                   
                                                                                                     | */                                                                                                  
                                                                                                     | public static &lt;T&gt; T[] validIndex(T[] array, int index, String message, Object... values) {    
                                                                                                     | Validate.notNull(array);                                                                            
                                                                                                     | if (index &lt; 0 || index &gt;= array.length) {                                                     
                                                                                                     | throw new IndexOutOfBoundsException(String.format(message, values));                                
                                                                                                     | }                                                                                                   
                                                                                                     | return array;                                                                                       
                                                                                                     | }
