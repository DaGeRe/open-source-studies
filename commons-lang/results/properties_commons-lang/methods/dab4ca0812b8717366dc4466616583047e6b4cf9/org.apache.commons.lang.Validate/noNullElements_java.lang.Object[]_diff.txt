/**                                                                                                  |                                                                                                     
* &lt;p&gt;Validate an argument, throwing &lt;code&gt;IllegalArgumentException&lt;/code&gt;          |                                                                                                     
* if the argument array has &lt;code&gt;null&lt;/code&gt; elements or is                             |                                                                                                     
* &lt;code&gt;null&lt;/code&gt;.&lt;/p&gt;                                                           |                                                                                                     
*                                                                                                    |                                                                                                     
* &lt;pre&gt;                                                                                        |                                                                                                     
* Validate.noNullElements(myArray);                                                                  |                                                                                                     
* &lt;/pre&gt;                                                                                       |                                                                                                     
*                                                                                                    |                                                                                                     
* &lt;p&gt;If the array has a null element the message in the exception is                           |                                                                                                     
* 'The validated array contains null element at index: '.&lt;/p&gt;                                  |                                                                                                     
*                                                                                                    |                                                                                                     
* &lt;p&gt;If the array is null then the message in the exception is 'The validated object is null'. |                                                                                                     
*                                                                                                    |                                                                                                     
* @param array  the array to check                                                                   |                                                                                                     
* @throws IllegalArgumentException if the array has &lt;code&gt;null&lt;/code&gt;                    |                                                                                                     
*  elements or is &lt;code&gt;null&lt;/code&gt;                                                      |                                                                                                     
*/                                                                                                   |                                                                                                     
public static void noNullElements(Object[] array) {                                                  |                                                                                                     
Validate.notNull(array);                                                                             |                                                                                                     
for (int i = 0; i &lt; array.length; i++) {                                                          |                                                                                                     
if (array[i] == null) {                                                                              |                                                                                                     
throw new IllegalArgumentException("The validated array contains null element at index: " + i);      |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |
