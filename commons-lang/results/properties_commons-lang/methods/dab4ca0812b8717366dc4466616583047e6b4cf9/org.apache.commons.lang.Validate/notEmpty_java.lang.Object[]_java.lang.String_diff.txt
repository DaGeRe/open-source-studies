/**                                                                                                  |                                                                                                     
* &lt;p&gt;Validate an argument, throwing &lt;code&gt;IllegalArgumentException&lt;/code&gt;          |                                                                                                     
* if the argument array is empty (&lt;code&gt;null&lt;/code&gt; or no elements).&lt;/p&gt;           |                                                                                                     
*                                                                                                    |                                                                                                     
* &lt;pre&gt;                                                                                        |                                                                                                     
* Validate.notEmpty(myArray, "The array must not be empty");                                         |                                                                                                     
* &lt;/pre&gt;                                                                                       |                                                                                                     
*                                                                                                    |                                                                                                     
* @param array  the array to check is not empty                                                      |                                                                                                     
* @param message  the exception message you would like to see if the array is empty                  |                                                                                                     
* @throws IllegalArgumentException if the array is empty                                             |                                                                                                     
*/                                                                                                   |                                                                                                     
public static void notEmpty(Object[] array, String message) {                                        |                                                                                                     
if (array == null || array.length == 0) {                                                            |                                                                                                     
throw new IllegalArgumentException(message);                                                         |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |
