/**
 * Create a new triple instance.
 *
 * @param left  the left value, may be null
 * @param middle  the middle value, may be null
 * @param right  the right value, may be null
 */
public MutableTriple(L left, M middle, R right) {
    super();
    this.left = left;
    this.middle = middle;
    this.right = right;
}
