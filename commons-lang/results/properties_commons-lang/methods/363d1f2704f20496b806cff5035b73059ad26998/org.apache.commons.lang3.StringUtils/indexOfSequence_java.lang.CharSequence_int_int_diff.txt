                                                                                                     | static int indexOfSequence(CharSequence cs, int searchChar, int start) {                            
                                                                                                     | if (cs instanceof String) {                                                                         
                                                                                                     | return ((String) cs).indexOf(searchChar, start);                                                    
                                                                                                     | } else {                                                                                            
                                                                                                     | int sz = cs.length();                                                                               
                                                                                                     | if (start &lt; 0) {                                                                                 
                                                                                                     | start = 0;                                                                                          
                                                                                                     | }                                                                                                   
                                                                                                     | for (int i = start; i &lt; sz; i++) {                                                               
                                                                                                     | if (cs.charAt(i) == searchChar) {                                                                   
                                                                                                     | return i;                                                                                           
                                                                                                     | }                                                                                                   
                                                                                                     | }                                                                                                   
                                                                                                     | return -1;                                                                                          
                                                                                                     | }
